###Strings
#####Indexing
if length is n then word[i] gives i'th charecter; where i belongs to -n to n-1 

word = "swaroop"

word[0] = 's' = word[-7]

word[6] = 'p' = word[-1]
#####Slicing
word[i:j] => i include but j exclude

word[:i] + word[i:]=> 'swaroop'

word[0:7] => 'swaroop'

word[-7:-1] => 'swaroo'

word[-7:-0] => ''

word[-7:] => 'swaroop'

word[0:] => 'swaroop'
